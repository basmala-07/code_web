OOP (Classes & Inheritance)

Source Code

class Student {
public:
  string name;
  Student(string n){ name = n; }
  void show(){ cout << name; }
};

class Animal {
public:
  void eat(){ cout << "Eating "; }
};
class Dog : public Animal {
public:
  void bark(){ cout << "Barking"; }
};

Student s("Mariam");
Dog d;
s.show();
d.eat();
d.bark();

Console Result

Mariam
Eating Barking
